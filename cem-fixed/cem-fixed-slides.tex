\begin{column}{0.42\textwidth}
\textbf{Defense Strategies:}

\vspace{0.15cm}
\small
\begin{enumerate}\setlength{\itemsep}{2pt}
    \item \textbf{Noise corruption}: Add Gaussian noise $z' = z + \varepsilon$
    \item \textbf{Distance correlation} (Nopeek): Minimize $\text{dCor}(x, z)$
    \item \textbf{Bottleneck}: Reduce channels (128 $\to$ 8)
    \item \textbf{ARL}: Adversarial representation learning
    \item \textbf{CEM} (this paper): Maximize conditional entropy $\mathcal{H}(x|z)$
\end{enumerate}

\vspace{0.15cm}
\begin{alertblock}{\small Key Insight}
\small CEM can be \textbf{plugged into any} defense strategy above as an add-on, consistently boosting privacy.
\end{alertblock}

\begin{frame}{Background: CEM --- Conditional Entropy Maximization}
\vspace{-0.15cm}

\begin{columns}[T]
\begin{column}{0.52\textwidth}
\textbf{Core Theory (Paper Theorem 1):}

\vspace{0.1cm}
\small
The reconstruction MSE has a lower bound related to conditional entropy:
\[ \xi^* \geq \frac{1}{2\pi e} \exp\!\left(\frac{2}{d}\,\mathcal{H}(x|z)\right) \]

where $\xi^*$ is the optimal reconstruction MSE and $\mathcal{H}(x|z)$ is the conditional entropy of inputs given features.

\vspace{0.1cm}
\textbf{Implication:} If we \textbf{maximize} $\mathcal{H}(x|z)$, the attacker's reconstruction MSE must increase $\Rightarrow$ better privacy.

\vspace{0.1cm}
\textbf{How to maximize $\mathcal{H}(x|z)$?}
\begin{itemize}\small\setlength{\itemsep}{0pt}
    \item Estimate feature distribution with mixture model
    \item Compute conditional entropy from cluster statistics
    \item Optimize via gradient backpropagation
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\textbf{Paper's GMM Approach:}

\vspace{0.1cm}
\small
\begin{enumerate}\setlength{\itemsep}{1pt}
    \item Fit $k$-component Gaussian mixture to $z$:
    \[ p(z) = \sum_{j=1}^{k} \pi_j \,\mathcal{N}(\mu_j, \Sigma_j) \]
    \item Hard assignment: $z \to \text{nearest } \mu_j$
    \item Update $\pi_j$, $\Sigma_j$ with running average
    \item CEM loss (maximizes entropy):
    \[ \mathcal{L}_C = \sum_{j} \pi_j\!\left(\!-\!\log\pi_j + \tfrac{1}{2}\log\tfrac{|\Sigma_j + \Sigma_p|}{|\Sigma_p|}\right) \]
    \item Joint optimization:
    \[ \mathcal{L} = \mathcal{L}_D + \lambda \cdot \mathcal{L}_C \]
\end{enumerate}
\vspace{0.05cm}
{\footnotesize $\Sigma_p$: noise covariance; $\lambda$: weight factor (default 16)}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{My Contribution: Replace GMM with SlotAttention}

\begin{block}{Problem with GMM-based CEM}
\small GMM uses \textbf{hard assignment} (each feature $\to$ nearest centroid). This is non-differentiable --- gradients must be approximated. The clustering quality depends heavily on initialization and component count $k$.
\end{block}

\vspace{0.15cm}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{\bad{Paper: GMM Hard Clustering}}
\small
\begin{itemize}\setlength{\itemsep}{2pt}
    \item Non-differentiable assignment step
    \item Running average update (stale statistics)
    \item Fixed $k$ components, sensitive to init
    \item No gradient through cluster assignment
    \item Covariance update is approximate
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{\fix{Mine: SlotAttention Soft Clustering}}
\small
\begin{itemize}\setlength{\itemsep}{2pt}
    \item \good{Fully differentiable} soft attention
    \item Iterative GRU refinement (learned dynamics)
    \item Adaptive slot count, robust to init
    \item \good{End-to-end gradient flow} to encoder
    \item Geometric variance (exact computation)
\end{itemize}
\end{column}
\end{columns}

% SLIDE 8: Bug 1

\begin{frame}{Bug 1: Loss Direction is Completely Reversed}

\begin{columns}[T]
\begin{column}{0.5\textwidth}
\textbf{\bad{Original formula:}}
\[ \mathcal{L} = \text{ReLU}\!\Big(\log(\theta) - \log(\bar{v} + \varepsilon)\Big) \]
\vspace{-0.1cm}
\begin{itemize}\small
    \item Positive when $\bar{v} < \theta$ (variance \textbf{below} threshold)
    \item Penalizes \textbf{low variance} $\Rightarrow$ drives variance \textbf{up}
    \item Encoder pushed to \textbf{spread} same-class features
    \item \bad{Opposite to CEM intent --- helps the attacker}
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\textbf{\fix{cem-fixed formula:}}
\[ \mathcal{L} = \text{ReLU}\!\Big(\log(\bar{v} + \gamma) - \log(\theta)\Big) \]
\vspace{-0.1cm}
\begin{itemize}\small
    \item Positive when $\bar{v} > \theta$ (variance \textbf{above} threshold)
    \item Penalizes \textbf{high variance} $\Rightarrow$ drives variance \textbf{down}
    \item Encoder pushed to \textbf{compress} features into tight sub-clusters
    \item \fix{Gradient direction matches CEM defense goal}
\end{itemize}
\vspace{0.1cm}
{\small Threshold: $\theta = \texttt{var\_thr} \times \texttt{reg\_str}^2 + \gamma$, $\;\gamma{=}0.01$}
\end{column}
\end{columns}

\vspace{0.1cm}
\begin{alertblock}{Impact}
\small With the original formula, increasing \texttt{lambd} would \textit{actively degrade} privacy. Every gradient step was working \textbf{against} the defense. This single bug renders the entire CEM mechanism counterproductive.
\end{alertblock}

\end{frame}

\begin{frame}{Engineering Fix 1: FeatureMemoryBank}
\vspace{-0.1cm}

\small\textbf{Problem}: FaceScrub has 530 classes. With batch size 256, each class appears only $\sim$0.5 times per batch. Without memory, most classes are \textbf{skipped entirely} in each step.

\vspace{0.05cm}
\begin{columns}[T]
\begin{column}{0.52\textwidth}
\textbf{FeatureMemoryBank design:}

\vspace{0.05cm}
\begin{tikzpicture}[scale=0.82,
    block/.style={rectangle, rounded corners, draw=teal!70, fill=teal!8,
                  minimum width=4.4cm, minimum height=0.45cm, align=center, font=\small}
]
\node[block] (b1) {bank: Tensor[530, 64, 128]};
\node[block, below=0.12cm of b1] (b2) {ptr: LongTensor[530] \small (ring buffer ptr)};
\node[block, below=0.12cm of b2] (b3) {valid: LongTensor[530] \small (fill count/class)};
\node[block, below=0.12cm of b3, fill=green!8, draw=green!60!black] (upd) {\code{update(feats, labels)} \small\code{@no\_grad}};
\node[block, below=0.12cm of upd, fill=green!8, draw=green!60!black] (get) {\code{get\_class\_features(cls)} $\to$ [n, 128]};
\draw[-{Stealth}, thick, gray] (b3) -- (upd);
\draw[-{Stealth}, thick, gray] (upd) -- (get);
\end{tikzpicture}
\end{column}
\begin{column}{0.45\textwidth}
\textbf{How it feeds SlotAttention:}

\vspace{0.05cm}
\small
\begin{enumerate}\setlength{\itemsep}{1pt}
    \item Each batch: write projected feats to bank (\code{no\_grad})
    \item For class $c$: read up to 64 historical feats from bank
    \item \code{cat([batch\_feats, bank\_feats])} $\to N_c$ samples
    \item SA receives $N_c = n_{\text{batch}} + n_{\text{bank}} \gg 2$
\end{enumerate}

\vspace{0.1cm}
\textbf{Gradient design:}
\begin{itemize}\small
    \item Bank feats: \textbf{no gradient} (detached history)
    \item Current batch feats: \textbf{has gradient} (drives encoder)
    \item $\Rightarrow$ CEM loss only updates current encoder weights
\end{itemize}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Engineering Fix 2: proj\_down for High-Dimensional Features}

\begin{frame}{Engineering Fix 3 \& 4: Detach Centroids + Deterministic Inference}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Fix 3: Detach centroids}

\vspace{0.1cm}
\small
Without \code{.detach()}, the gradient has \textbf{two paths}:
\begin{enumerate}
    \item Features move toward centroid \good{(correct)}
    \item Centroid moves toward features \bad{(shortcut!)}
\end{enumerate}

Path 2 satisfies the loss with \textbf{zero encoder change}. The optimizer takes the easy path.

\vspace{0.1cm}
\textbf{Fix:} $c^* = \text{einsum}(\text{attn}, f)$\code{.detach()}

\vspace{0.05cm}
$\Rightarrow$ Only path 1 remains. Encoder \textbf{must genuinely} compress features toward fixed centroids.

\vspace{0.15cm}
\begin{tikzpicture}[scale=0.65,
    block/.style={rectangle, rounded corners, draw=gray, fill=gray!8,
                  minimum width=3.5cm, minimum height=0.4cm, align=center, font=\scriptsize}
]
\node[block, fill=green!8, draw=green!60!black] (feats) {features $f$ (has grad)};
\node[block, below=0.2cm of feats, fill=teal!8, draw=teal!60] (cent) {$c^* = \text{einsum}(\text{attn},f)$\,\code{.detach()}};
\node[block, below=0.2cm of cent, fill=green!8, draw=green!60!black] (diff) {loss $= \text{attn} \cdot (f - c^*)^2$};
\draw[-{Stealth}, thick, gray] (feats) -- (cent);
\draw[-{Stealth}, thick, gray] (cent) -- (diff);
\draw[-{Stealth[length=2mm]}, thick, green!60!black] (feats.east) to[out=30,in=30]
    node[right, font=\tiny, text=goodgreen] {only path} (diff.east);
\end{tikzpicture}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Fix 4: Deterministic inference}

\vspace{0.1cm}
\small
\textbf{Problem:} SlotAttention initializes slots from $\mu + \sigma \cdot \varepsilon$, $\varepsilon \sim \mathcal{N}(0,1)$. During MIA evaluation, each forward pass gives \textbf{different} attention distributions $\Rightarrow$ results not reproducible.

\vspace{0.15cm}
\textbf{Fix:}
\begin{itemize}\small\setlength{\itemsep}{2pt}
    \item \textbf{Training}: $\text{slots} = \mu + \sigma \cdot \varepsilon$ (stochastic, preserves exploration)
    \item \textbf{Eval}: $\text{slots} = \mu$ (deterministic, reproducible)
\end{itemize}

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}[scale=0.65,
    block/.style={rectangle, rounded corners, draw=gray, fill=gray!8,
                  minimum width=3.5cm, minimum height=0.4cm, align=center, font=\scriptsize}
]
\node[block] (mu) {learnable $\mu$, $\sigma$};
\node[block, below=0.15cm of mu, fill=green!8, draw=green!60!black] (tr) {Train: $\mu + \sigma\varepsilon$ (stochastic)};
\node[block, below=0.15cm of tr, fill=teal!8, draw=teal!60] (ev) {Eval: $\mu$ only (deterministic)};
\draw[-{Stealth}, thick, gray] (mu) -- (tr);
\draw[-{Stealth}, thick, gray] (mu) -- (ev);
\end{tikzpicture}
\end{center}

$\Rightarrow$ MIA metrics are \fix{fully reproducible} across runs.
\end{column}
\end{columns}

\end{frame}

\begin{frame}{SlotAttention Mechanism: Step-by-Step}

\vspace{-0.1cm}
\begin{columns}[T]
\begin{column}{0.52\textwidth}
\textbf{Algorithm (per class $c$, each training step):}

\vspace{0.1cm}
\scriptsize
\begin{enumerate}\setlength{\itemsep}{2pt}
    \item \textbf{Input}: $N_c$ features from batch + memory bank
    \[ \text{inputs} = [f_1, f_2, \ldots, f_{N_c}] \in \mathbb{R}^{N_c \times D} \]
    \item \textbf{Init slots}: $S$ learnable prototypes
    \[ s_j^{(0)} = \mu_j + \sigma_j \cdot \varepsilon_j, \quad j=1,\ldots,S \]
    \item \textbf{For $t = 1, \ldots, T$ iterations:}
    \begin{enumerate}\scriptsize
        \item Compute attention: $A_{ij} = \frac{\exp(q_j \cdot k_i / \sqrt{D})}{\sum_j \exp(q_j \cdot k_i / \sqrt{D})}$
        \item Normalize over inputs: $\hat{A}_{ij} = A_{ij} / \sum_i A_{ij}$
        \item Weighted mean: $u_j = \sum_i \hat{A}_{ij} \cdot v(f_i)$
        \item GRU update: $s_j^{(t)} = \text{GRU}(u_j, s_j^{(t-1)})$
        \item MLP residual: $s_j^{(t)} \mathrel{+}= \text{MLP}(s_j^{(t)})$
    \end{enumerate}
    \item \textbf{Output}: refined slots $s_j^{(T)}$ + attention weights $\hat{A}$
\end{enumerate}
\end{column}
\begin{column}{0.45\textwidth}
\textbf{How it differs from GMM:}

\vspace{0.1cm}
\small
\begin{tabular}{p{1.8cm}p{1.5cm}p{1.5cm}}
\toprule
& \textbf{GMM} & \textbf{SlotAttn} \\
\midrule
Assignment & Hard & \good{Soft} \\
Update & Running avg & \good{GRU} \\
Gradient & Approx & \good{Exact} \\
Iterations & 1 & $T$ (3--5) \\
Init & Random & Learned \\
\bottomrule
\end{tabular}

\vspace{0.15cm}
\textbf{After SlotAttention:}
\scriptsize
\begin{enumerate}\setlength{\itemsep}{1pt}
    \item Compute centroids: $c_j = \sum_i \hat{A}_{ij} \cdot f_i$
    \item \code{.detach()} centroids (Fix 3)
    \item Geometric variance per slot:\\
    $v_j = \frac{\sum_i \hat{A}_{ij} \cdot \|f_i - c_j^*\|^2}{\sum_i \hat{A}_{ij} \cdot D}$
    \item Mean variance: $\bar{v} = \frac{1}{S}\sum_j v_j$
    \item CEM loss: $\mathcal{L} = \text{ReLU}(\log(\bar{v}+\gamma) - \log\theta)$
\end{enumerate}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Why SlotAttention CEM Outperforms GMM CEM}

\vspace{-0.15cm}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{\bad{GMM CEM (Paper):}}

\vspace{0.05cm}
\small
\begin{enumerate}\setlength{\itemsep}{2pt}
    \item \textbf{Hard assignment} $\to$ non-differentiable\\
    Gradients must be approximated; encoder receives \textbf{noisy, imprecise} signals
    \item \textbf{Running average} covariance update\\
    Statistics are stale; does not reflect current feature distribution
    \item \textbf{Fixed $k$ components}\\
    Sensitive to initialization; may not match true cluster structure
    \item \textbf{No iterative refinement}\\
    Single-pass assignment; errors cannot be corrected
\end{enumerate}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{\fix{SlotAttention CEM (Mine):}}

\vspace{0.05cm}
\small
\begin{enumerate}\setlength{\itemsep}{2pt}
    \item \textbf{Soft attention} $\to$ fully differentiable\\
    Encoder receives \good{smooth, informative} gradients via attention weights
    \item \textbf{GRU-based} centroid update\\
    \good{Learned dynamics}; adapts to feature distribution in real time
    \item \textbf{Learned slot initialization}\\
    $\mu, \sigma$ trainable; slots adapt to the task during training
    \item \textbf{$T$-step iterative refinement}\\
    Each iteration \good{improves} assignments; self-correcting
\end{enumerate}
\end{column}
\end{columns}

\vspace{0.05cm}
\begin{block}{\small Bottom Line}
\small Soft attention $\Rightarrow$ smoother gradients $\Rightarrow$ tighter sub-clusters $\Rightarrow$ attacker's reconstruction error increases \good{+22.3\%}. {\scriptsize\color{warnorg}(results from early implementation, updated experiments in progress)}
\end{block}
