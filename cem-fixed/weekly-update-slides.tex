\begin{frame}{No-Defense Baseline --- Setup}

\begin{itemize}\setlength{\itemsep}{8pt}
    \item \textbf{Purpose}: Establish the upper bound for accuracy and lower bound for privacy
    \item \textbf{Configuration}: Same architecture (VGG-11), no noise, no CEM, no regularization
    \item All defense mechanisms disabled:
    \begin{itemize}
        \item Gaussian noise $\sigma = 0$
        \item CEM weight $\lambda = 0$
        \item AT regularization = None
        \item Bottleneck = None
    \end{itemize}
    \item \textbf{Dataset}: FaceScrub (530 classes, 64$\times$64)
    \item \textbf{Training}: 240 epochs
\end{itemize}

\end{frame}

% ============================================================================
\begin{frame}{No-Defense Baseline --- Results}

Dec Infer PSNR $\downarrow$ & 31.76 & --- \\

\vspace{0.3cm}

\begin{alertblock}{Interpretation}
\begin{itemize}\small
    \item Accuracy \textbf{89.43\%} confirms VGG-11 has strong classification capability on FaceScrub
    \item MSE = 0.00067, SSIM = 0.949 $\Rightarrow$ attacker can \bad{nearly perfectly reconstruct} faces
    \item This is the \textbf{worst-case privacy scenario} --- any defense must improve upon this
\end{itemize}
\end{alertblock}

\begin{frame}{Motivation --- Accuracy vs Privacy Trade-off}

\begin{itemize}\setlength{\itemsep}{6pt}
    \item Previous best (exp13): Acc = 78.15\%, Dec Infer MSE = 0.0258
    \item Paper best (Noise\_ARL+CEM): Acc = \textbf{80.33\%}, MSE = 0.0211
    \item \textbf{Gap}: My accuracy is 2.2\% lower than paper
    \item \textbf{Goal}: Push accuracy closer to 80\% while keeping privacy advantage
\end{itemize}

\vspace{0.3cm}

\begin{block}{Strategy: Two Changes}
\begin{enumerate}\setlength{\itemsep}{4pt}
    \item \textbf{Wider bottleneck}: C8 $\to$ C16 (8 $\to$ 16 channels)\\
    {\small More information capacity through the split point, 87.5\% compression vs 93.75\%}
    \item \textbf{Weaker defense parameters}: Lower $\sigma$, $\lambda$, AT strength\\
    {\small Trade a small amount of privacy for better classification}
\end{enumerate}
\end{block}

\end{frame}
